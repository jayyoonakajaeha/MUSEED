' Boundaries
' ##################################
!unquoted procedure Container_Boundary($alias, $label, $tags="", $link="", $descr = "")
    !if ($tags != "")
        !$allTags = $tags + "+container"
    !else
        !$allTags = "container"
    !endif
    ' $type defined via $tag style
    Boundary($alias, $label, "", $allTags, $link, $descr)
!endprocedure

' Define System Boundary
System_Boundary(museed_system, "MUSEED: AI-Powered Music Discovery Platform") {

    ' Define Containers (Major Components)
    Container(frontend, "Frontend", "React / Next.js", "User Interface for music discovery and playlist management")
    Container(backend, "Backend API", "Python / FastAPI", "Handles user authentication, API endpoints, and business logic")
    Container(aiml_service, "AI / ML Service", "PyTorch / FAISS", "Extracts audio embeddings and performs similarity search")
    Container(postgresql_db, "PostgreSQL Database", "SQL Database", "Stores user data, playlist metadata, and track information")
    Container(faiss_db, "FAISS Embeddings DB", "Vector Database", "Stores high-dimensional audio embeddings for fast similarity search")
    Container(file_storage, "File Storage", "Disk / Cloud Storage", "Stores uploaded audio files")

    ' Define Relationships
    Rel(frontend, backend, "Makes API Calls", "HTTP/REST")
    Rel(backend, file_storage, "Uploads / Downloads Audio", "HTTP")
    Rel(backend, aiml_service, "Requests AI Processing", "HTTP/REST")
    Rel(backend, postgresql_db, "Reads/Writes Metadata", "SQL")
    Rel(aiml_service, faiss_db, "Reads/Writes Embeddings", "Vector Search Protocol")
    Rel(aiml_service, file_storage, "Reads Audio for Embedding", "File Access")

    ' External User
    Person(user, "User", "Music enthusiast, creator, or producer")

    ' User interacts with Frontend
    Rel(user, frontend, "Uses", "Web Browser")
}
